    <script>
        // Timer functionality - Countdown from 15 to 0
        let countdownValue = 15;
        let countdownActive = true;
        const timerContainer = document.querySelector('.timer-container');
        const timerElement = document.getElementById('timer');
        
        function updateTimer() {
            if (countdownActive) {
                timerElement.textContent = countdownValue;
                
                if (countdownValue > 0) {
                    countdownValue--;
                } else {
                    // When timer reaches 0
                    countdownActive = false;
                    showBirthdayPopup();
                    
                    // Restart the countdown after 5 seconds
                    setTimeout(() => {
                        countdownValue = 15;
                        countdownActive = true;
                        timerElement.style.display = 'block';
                    }, 5000);
                }
            }
        }
        
        function showBirthdayPopup() {
            // Hide the timer
            timerElement.style.display = 'none';
            
            // Create and show the birthday message
            const popup = document.createElement('div');
            popup.className = 'birthday-popup';
            popup.textContent = 'Enjoy your day Cutipie!!';
            timerContainer.appendChild(popup);
            
            // Remove popup after 5 seconds
            setTimeout(() => {
                popup.remove();
            }, 5000);
        }
        
        setInterval(updateTimer, 1000);
        
        // Image Modal functionality
        const modal = document.getElementById("imageModal");
        const modalImg = document.getElementById("modalImage");
        const captionText = document.getElementById("modalCaption");
        
        function openModal(element) {
            const imageSrc = element.querySelector('.memory-img').src;
            const title = element.querySelector('.memory-title').textContent;
            const subtitle = element.querySelector('.memory-subtitle').textContent;
            
            modal.style.display = "flex";
            modalImg.src = imageSrc;
            
            // Create caption with both title and subtitle
            captionText.innerHTML = `
                <div class="modal-title">${title}</div>
                <div class="modal-subtitle">${subtitle}</div>
            `;
            
            document.body.style.overflow = "hidden"; // Prevent scrolling
        }
        
        function closeModal() {
            modal.style.display = "none";
            document.body.style.overflow = "auto"; // Enable scrolling
        }
        
        // Close modal when clicking outside the image
        modal.addEventListener('click', function(event) {
            if (event.target === modal) {
                closeModal();
            }
        });
        
        // Message functionality - Store all messages in localStorage
        let messages = [];
        let editingMessageId = null;
        
        // Load messages from localStorage
        function loadMessagesFromStorage() {
            const storedMessages = localStorage.getItem('birthdayMessages');
            if (storedMessages) {
                messages = JSON.parse(storedMessages);
            }
            updateMessageDisplay();
        }
        
        // Save messages to localStorage
        function saveMessagesToStorage() {
            localStorage.setItem('birthdayMessages', JSON.stringify(messages));
        }
        
        // Generate a unique ID for each message
        function generateId() {
            return Date.now().toString(36) + Math.random().toString(36).substr(2);
        }
        
        // Format the current time
        function getCurrentTime() {
            const now = new Date();
            return now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
        }
        
        // Update the message display
        function updateMessageDisplay() {
            const messageDisplay = document.getElementById('messageDisplay');
            
            if (messages.length === 0) {
                messageDisplay.innerHTML = '<div class="empty-message">No messages yet. Be the first to wish!</div>';
                return;
            }
            
            messageDisplay.innerHTML = '';
            messages.forEach(message => {
                const messageElement = document.createElement('div');
                messageElement.className = 'message-item';
                messageElement.dataset.id = message.id;
                
                messageElement.innerHTML = `
                    <div class="message-text">${message.text}</div>
                    <div class="message-time">${message.time}</div>
                    <div class="message-actions">
                        <button class="edit-message-btn" onclick="editMessage('${message.id}')">
                            <i class="fas fa-edit"></i>
                        </button>
                        <button class="delete-message-btn" onclick="deleteMessage('${message.id}')">
                            <i class="fas fa-trash"></i>
                        </button>
                    </div>
                `;
                
                messageDisplay.appendChild(messageElement);
            });
            
            // Scroll to the bottom to show the latest message
            messageDisplay.scrollTop = messageDisplay.scrollHeight;
        }
        
        // Add a new message
        function addMessage(text) {
            const newMessage = {
                id: generateId(),
                text: text,
                time: getCurrentTime()
            };
            
            messages.push(newMessage);
            saveMessagesToStorage();
            updateMessageDisplay();
        }
        
        // Edit an existing message
        function editMessage(id) {
            const message = messages.find(m => m.id === id);
            if (message) {
                document.getElementById('messageInput').value = message.text;
                editingMessageId = id;
                document.getElementById('sendMessageBtn').innerHTML = '<i class="fas fa-save"></i> Update Message';
            }
        }
        
        // Update an existing message
        function updateMessage(id, newText) {
            const messageIndex = messages.findIndex(m => m.id === id);
            if (messageIndex !== -1) {
                messages[messageIndex].text = newText;
                messages[messageIndex].time = getCurrentTime() + ' (edited)';
                saveMessagesToStorage();
                updateMessageDisplay();
            }
        }
        
        // Delete a message
        function deleteMessage(id) {
            messages = messages.filter(m => m.id !== id);
            saveMessagesToStorage();
            updateMessageDisplay();
        }
        
        // Clear all messages
        function clearAllMessages() {
            if (confirm('Are you sure you want to clear all messages? This cannot be undone.')) {
                messages = [];
                saveMessagesToStorage();
                updateMessageDisplay();
            }
        }
        
        // Button functionality
        document.getElementById('confettiBtn').addEventListener('click', createConfetti);
        
        document.getElementById('sendMessageBtn').addEventListener('click', function() {
            const messageInput = document.getElementById('messageInput');
            const message = messageInput.value.trim();
            
            if (message) {
                if (editingMessageId) {
                    // Update existing message
                    updateMessage(editingMessageId, message);
                    editingMessageId = null;
                    this.innerHTML = '<i class="fas fa-paper-plane"></i> Send Message';
                } else {
                    // Add new message
                    addMessage(message);
                }
                
                messageInput.value = '';
            }
        });
        
        document.getElementById('clearMessagesBtn').addEventListener('click', clearAllMessages);
        
        // Confetti creation
        function createConfetti() {
            const container = document.querySelector('.container');
            const colors = ['#ff5e5e', '#5e7aff', '#5eff7a', '#ffdb5e', '#db5eff'];
            
            for (let i = 0; i < 50; i++) {
                const confetti = document.createElement('div');
                confetti.className = 'confetti';
                confetti.style.left = Math.random() * 100 + '%';
                confetti.style.top = Math.random() * 100 + '%';
                confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                confetti.style.width = Math.random() * 15 + 5 + 'px';
                confetti.style.height = confetti.style.width;
                container.appendChild(confetti);
                
                // Animate confetti
                const animationDuration = Math.random() * 2 + 1 + 's';
                confetti.style.transition = `transform ${animationDuration}, opacity ${animationDuration}`;
                
                setTimeout(() => {
                    confetti.style.transform = `translate(${Math.random() * 200 - 100}px, ${Math.random() * 200 - 100}px) rotate(${Math.random() * 360}deg)`;
                    confetti.style.opacity = '0';
                }, 10);
                
                // Remove confetti after animation
                setTimeout(() => {
                    confetti.remove();
                }, parseFloat(animationDuration) * 1000 + 1000);
            }
        }
        
        // Initialize the page
        window.addEventListener('load', function() {
            // Load messages from localStorage
            loadMessagesFromStorage();
            
            // Add some initial confetti
            setTimeout(createConfetti, 500);
            setTimeout(createConfetti, 1000);
            setTimeout(createConfetti, 1500);
        });
